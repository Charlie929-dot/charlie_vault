#include <iostream>
#include <string>
#include <math.h>
using namespace std;


typedef struct dots{
	double x;
	double y;
}sd;
//以矩形中心为圆心，取一点（x,y）置于第一象限，使得矩形在坐标系中正置
typedef struct squares{
	sd dot;
}ss;

string location(sd dot,ss square){
	bool onx=abs(dot.x)==abs(square.dot.x),inx=abs(dot.x)<abs(square.dot.x),overx=abs(dot.x)>abs(square.dot.x),
		 ony=abs(dot.y)==abs(square.dot.y),iny=abs(dot.y)<abs(square.dot.y),overy=abs(dot.y)>abs(square.dot.y);
	if(onx&&ony) return "在角上";
	
	if((onx&&iny)||(inx&&ony)) return "在边上";
	if(inx&&iny) return "在矩形内";
	if(overx&&overy) return "在矩形外";
	
} 
int main(){
	sd dot;
	ss square;
	cin>>dot.x>>dot.y>>square.dot.x>>square.dot.y;
	cout<<location(dot,square)<<endl;
	return 0;
}
